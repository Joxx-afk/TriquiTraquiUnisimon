def obtener_mejor_movimiento(tablero, dificultad):
    if dificultad == 'facil':
        return random.choice(obtener_movimientos_disponibles(tablero))

    elif dificultad == 'medio':
        movimiento = encontrar_jugada_ganadora_o_bloqueo(tablero)
        if movimiento: return movimiento
        return random.choice(obtener_movimientos_disponibles(tablero))

    elif dificultad == 'dificil':
        mejor_puntaje = -math.inf
        mejor_movimiento = -1
        for movimiento in obtener_movimientos_disponibles(tablero):
            tablero[movimiento] = 'O'  # Probar jugada IA
            puntaje = minimax(tablero, profundidad=0, es_maximizador=False)
            tablero[movimiento] = ''   # Deshacer jugada
            if puntaje > mejor_puntaje:
                mejor_puntaje = puntaje
                mejor_movimiento = movimiento
        return mejor_movimiento

def minimax(tablero, profundidad, es_maximizador):
    if verificar_ganador(tablero, 'O'): return 10 - profundidad
    if verificar_ganador(tablero, 'X'): return profundidad - 10
    if es_empate(tablero): return 0

    if es_maximizador:
        mejor = -math.inf
        for m in obtener_movimientos_disponibles(tablero):
            tablero[m] = 'O'
            mejor = max(mejor, minimax(tablero, profundidad+1, False))
            tablero[m] = ''
        return mejor
    else:
        mejor = math.inf
        for m in obtener_movimientos_disponibles(tablero):
            tablero[m] = 'X'
            mejor = min(mejor, minimax(tablero, profundidad+1, True))
            tablero[m] = ''
        return mejor
